<script>
    // State machine for tasks
    const STATES = ['state-untouched', 'state-started', 'state-done'];
    const SCORE_EL = document.getElementById('headerScore');
    let currentScore = 72; // Starting state based on prompt

    // Initialize all tasks
    document.querySelectorAll('.task-row').forEach(row => {
        row.addEventListener('click', function() {
            // Find current state
            let currentStateIdx = 0;
            STATES.forEach((state, idx) => {
                if (this.classList.contains(state)) currentStateIdx = idx;
            });

            // Remove old state
            this.classList.remove(STATES[currentStateIdx]);

            // Calculate new state (cycle forward)
            let newStateIdx = (currentStateIdx + 1) % STATES.length;
            this.classList.add(STATES[newStateIdx]);

            // Animate Tick Spring Pop
            const tick = this.querySelector('.tick-circle');
            tick.classList.remove('pop-anim');
            void tick.offsetWidth; // Trigger reflow to restart animation
            tick.classList.add('pop-anim');

            // Trigger Confetti if entering 'Done'
            if (STATES[newStateIdx] === 'state-done') {
                triggerConfetti(tick);
                updateScore(1);
            } else if (STATES[currentStateIdx] === 'state-done') {
                updateScore(-1); // Reverted from done
            }
        });
    });

    function triggerConfetti(parentEl) {
        const wrapper = document.createElement('div');
        wrapper.className = 'confetti-wrapper';
        
        // Colors from our pastel palette
        const colors = ['#C8E6C9', '#FFCCBC', '#E1BEE7', '#BBDEFB', '#43A047'];
        
        for (let i = 0; i < 8; i++) {
            const particle = document.createElement('div');
            particle.className = 'confetti-particle';
            particle.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
            
            // Random direction
            const angle = Math.random() * Math.PI * 2;
            const distance = 20 + Math.random() * 30; // 20px to 50px spread
            const dx = Math.cos(angle) * distance;
            const dy = Math.sin(angle) * distance;
            
            particle.style.setProperty('--dx', `${dx}px`);
            particle.style.setProperty('--dy', `${dy}px`);
            
            // Randomize speed slightly
            const duration = 0.5 + Math.random() * 0.3;
            particle.style.animation = `burst ${duration}s ease-out forwards`;
            
            wrapper.appendChild(particle);
        }
        
        parentEl.appendChild(wrapper);
        
        // Cleanup DOM after animation
        setTimeout(() => {
            if(parentEl.contains(wrapper)) parentEl.removeChild(wrapper);
        }, 1000);
    }

    // Score updating logic (Visual fluff)
    function updateScore(change) {
        currentScore += (change * 4); // Arbitrary increment for demo
        if(currentScore > 100) currentScore = 100;
        
        // Width expansion visual trick
        SCORE_EL.style.transform = "scale(1.1)";
        setTimeout(() => { SCORE_EL.style.transform = "scale(1)"; }, 200);
        SCORE_EL.innerText = `Score: ${currentScore}%`;
        
        // Update bottom card too
        document.querySelector('.end-summary h2').innerText = `${currentScore}%`;
    }

    // Dismiss carry-forward banner
    function dismissBanner() {
        const banner = document.getElementById('carryForwardBanner');
        banner.style.transition = "all 0.4s ease";
        banner.style.opacity = "0";
        banner.style.transform = "translateX(50px)";
        
        setTimeout(() => {
            banner.style.display = 'none';
        }, 400);
    }

    // Bottom Nav Active State Switching
    function switchNav(el) {
        document.querySelectorAll('.nav-item').forEach(nav => nav.classList.remove('active'));
        el.classList.add('active');
    }
